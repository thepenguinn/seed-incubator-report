%% <name> <corner> <oppCorner>
\newcommand\innerEnclosureNode[3] {
    \draw
    let \p{TEMPDIMA} = ($(#2)-(#3)$),
    \n{TEMPDIMX} = {veclen(\x{TEMPDIMA}, 0)},
    \n{TEMPDIMY} = {veclen(0, \y{TEMPDIMA})}
    in
    ($(#2)!0.50!(#3)$)
    node (#1) [
        innerEnclosure,
        minimum height = \n{TEMPDIMY},
        minimum width = \n{TEMPDIMX},
        anchor = center,
    ] {}
    ;
}
