\def\absMainFourWidth{3}
\def\absMainFourHeight{6}

\subtikzpicturedef{subAbsMainFour} {
    l0, l1, l2, l3,
    b0, b1, r0,
    north, south, east, west,
    northeast, northwest, southeast, southwest,
    center,
    origin%
} {
    \draw (#1-start) coordinate (#1-origin);
    \subAbsCtkDiagramBase {#1-absCtkDiagramBase} {#1-origin} {origin}

    \draw [absCtkAuxModule]
    (#1-origin) coordinate (#1-bottomLeftCorner)
    (#1-bottomLeftCorner) ++(\absMainFourWidth, \absMainFourHeight)
    coordinate (#1-topRightCorner)

    \markgeocoordinate {#1}
    {(#1-topRightCorner)} {(#1-bottomLeftCorner)}
    {(#1-bottomLeftCorner)} {(#1-topRightCorner)}

    (#1-northwest) ++(-\absCtkPinPad, 0)
    ++(0, -{\absMainFourHeight / 8}) coordinate (#1-l0)
    ++(0, -{\absMainFourHeight / 4}) coordinate (#1-l1)
    ++(0, -{\absMainFourHeight / 4}) coordinate (#1-l2)
    ++(0, -{\absMainFourHeight / 4}) coordinate (#1-l3)

    (#1-southwest) ++(0.5, -\absCtkPinPad)
    coordinate (#1-b0)
    ++(0.5, 0) coordinate (#1-b1)

    (#1-northeast) ++(\absCtkPinPad, -0.5)
    coordinate (#1-r0)

    (#1-bottomLeftCorner) rectangle (#1-topRightCorner)

    (#1-center) coordinate (#1-mainCenter)

    \markgeocoordinate {#1}
    {(#1-north)} {(#1-b0)}
    {(#1-l0)} {(#1-r0)}

    ;

}

\subtikzpictureactivate{subAbsMainFour}
